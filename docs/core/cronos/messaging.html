 10    def __init__(self, driver, wait_time=10):
 11        """
 12        Inicializa o objeto WhatsAppMessenger com um driver do Selenium e tempo de espera padrão.
 13        
 14        :param driver: Instância do webdriver do Selenium
 15        :param wait_time: Tempo máximo para espera explícita de elementos (em segundos)
 16        """
 17        self.driver = driver
 18        self.wait_time = wait_time
 19        self.logger = logging.getLogger(self.__class__.__name__)

 22    def open_chat(self, contact_name):
 23        """
 24        Abre o chat com o contato especificado.
 25
 26        :param contact_name: Nome do contato conforme exibido na interface do WhatsApp.
 27        :raises Exception: Se o chat não puder ser aberto.
 28        """
 29        try:
 30            # Limpa a barra de pesquisa
 31            search_box = self.driver.find_element(By.XPATH, settings.CAIXA_RESEARCH_CONTACT)
 32            search_box.send_keys(Keys.CONTROL + 'a')
 33            search_box.send_keys(Keys.DELETE)
 34            
 35            # Digita o nome do contato
 36            search_box.send_keys(contact_name)
 37            self.logger.info(f"Buscando contato: {contact_name}")
 38            
 39            # Aguarda que o resultado da pesquisa apareça
 40            wait = WebDriverWait(self.driver, self.wait_time)
 41            search_result_xpath = settings.FIND_CONTACT.format(contact_name=contact_name)
 42            chat = wait.until(EC.presence_of_element_located((By.XPATH, search_result_xpath)))
 43            
 44            try:
 45                chat.click()
 46            except Exception as click_error:
 47                self.logger.warning(f"Click normal falhou: {click_error}. Tentando via JavaScript.")
 48                self.driver.execute_script("arguments[0].click();", chat)
 49            
 50            settings.time.sleep(3)
 51            self.logger.info(f"Chat aberto para: {contact_name}")
 52        except Exception as e:
 53            self.logger.error(f"Erro ao abrir chat para {contact_name}: {e}")
 54            raise

 56    def open_chat_non_contact(self, phone_number: str):
 57        """
 58        Abre a conversa para um número que não está na lista de contatos.
 59
 60        Este método clica no botão "Nova conversa", preenche o campo com o número
 61        e confirma para abrir o chat.
 62
 63        :param phone_number: Número do telefone (com código do país, ex: 5511999998888)
 64        :raises Exception: Se ocorrer erro ao abrir a conversa.
 65        """
 66        try:
 67            wait = WebDriverWait(self.driver, self.wait_time)
 68            # Clica no botão "Nova conversa" usando o atributo aria-label (ou outro seletor definido)
 69            new_chat_button = wait.until(
 70                EC.element_to_be_clickable((By.XPATH, settings.NEW_CHAT_NEXT_BUTTON))
 71            )
 72            new_chat_button.click()
 73            self.logger.info("Botão 'Nova conversa' clicado.")
 74
 75            # Aguarda o campo para digitar o número
 76            phone_input = wait.until(
 77                EC.presence_of_element_located((By.XPATH, settings.NEW_CHAT_PHONE_INPUT))
 78            )
 79            phone_input.send_keys(phone_number)
 80            self.logger.info(f"Número digitado: {phone_number}")
 81            settings.time.sleep(2)
 82            
 83            # Pressiona ENTER para confirmar o número e iniciar o chat
 84            phone_input.send_keys(Keys.RETURN)
 85            self.logger.info("ENTER pressionado para confirmar o número.")
 86            settings.time.sleep(2)
 87        except Exception as e:
 88            self.logger.error(f"Erro ao abrir chat para número não contato: {e}")
 89            raise

 91    def send_message(self, message):
 92        """
 93        Envia uma mensagem para o chat atualmente aberto.
 94        
 95        :param message: Texto da mensagem a ser enviada
 96        :raises Exception: Se houver erro no envio da mensagem
 97        """
 98        try:
 99            wait = WebDriverWait(self.driver, self.wait_time)
100            message_box = wait.until(EC.presence_of_element_located((By.XPATH, settings.MESSAGE_TEXT_BOX)))
101            message_box.send_keys(message)
102            settings.time.sleep(1)
103            message_box.send_keys(Keys.RETURN)
104            settings.time.sleep(20)
105            self.logger.info("Mensagem enviada com sucesso.")
106        except Exception as e:
107            self.logger.error(f"Erro ao enviar mensagem: {e}")
108            raise
