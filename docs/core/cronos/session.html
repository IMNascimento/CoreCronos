 20    """
 21    Gerencia a sessão do WhatsApp Web utilizando Selenium, com suporte a:
 22      - Salvar e carregar cookies e metadados (incluindo proxy)
 23      - Aplicar VPN se necessário (simulado)
 24      - Login e logout do WhatsApp Web
 25      - Configuração do driver para parecer uma aplicação legítima
 26      - Alteração de proxy e destruição da sessão
 27    
 28    Cada sessão é associada a um identificador (número ou nome) e os dados são salvos
 29    em um diretório dedicado para esse identificador.
 30    """
 31    
 32    def __init__(self, phone_number: str, use_vpn: bool = False, proxy: Optional[str] = None) -> None:
 33        """
 34        Inicializa a sessão do WhatsApp.
 35        
 36        :param phone_number: Número ou identificador associado à sessão.
 37        :param use_vpn: Se True, tenta aplicar VPN (simulado).
 38        :param proxy: Proxy a ser usado (ex.: "http://user:pass@proxy_server:port"). Se None, sem proxy.
 39        """
 40        self.phone_number: str = phone_number
 41        self.use_vpn: bool = use_vpn
 42        self.proxy: Optional[str] = proxy
 43        self.profile_path: Path = Path(settings.COOKIE_DIR) / phone_number
 44        self.driver: Optional[webdriver.Chrome] = None
 45        self.metadata: Dict[str, Any] = {}
 46        self.wait_time: int = 50  # Tempo máximo para espera explícita de elementos (em segundos)
 47        self._load_metadata()
 48
 49    def _apply_vpn(self) -> None:
 50        """
 51        Aplica configurações de VPN se habilitado.
 52        Essa implementação é simulada. Em uma implementação real, utilize chamadas a
 53        bibliotecas ou comandos externos.
 54        """
 55        if self.use_vpn and settings.VPN_CONFIG.get("use_vpn"):
 56            log_info(f"Aplicando VPN para o número {self.phone_number}")
 57            # Exemplo: subprocess.run(["vpnclient", "connect", settings.VPN_CONFIG["vpn_server"], ...])
 58            sleep(1)
 59
 60    def _get_chrome_options(self) -> Options:
 61        """
 62        Configura as opções do Chrome para a sessão, incluindo perfil, proxy e ajustes
 63        para disfarçar a automação.
 64        
 65        :return: Instância configurada de Options.
 66        """
 67        options = Options()
 68        options.add_argument(f"--user-data-dir={self.profile_path}")
 69        options.add_argument("--profile-directory=Default")
 70        # Configurações para parecer uma aplicação legítima:
 71        options.add_experimental_option("excludeSwitches", ["enable-automation"])
 72        options.add_experimental_option("useAutomationExtension", False)
 73        options.add_argument("--disable-blink-features=AutomationControlled")
 74        # Se um proxy estiver definido, adiciona a opção:
 75        if self.proxy:
 76            options.add_argument(f'--proxy-server={self.proxy}')
 77        return options
 78
 79    def _setup_driver(self) -> None:
 80        """
 81        Configura e inicia o driver do Chrome com as opções necessárias.
 82        """
 83        self._apply_vpn()
 84        options = self._get_chrome_options()
 85        try:
 86            self.driver = webdriver.Chrome(
 87                service=Service(ChromeDriverManager().install()),
 88                options=options
 89            )
 90            # Remover a flag de automação
 91            self.driver.execute_script("Object.defineProperty(navigator, 'webdriver', {get: () => undefined});")
 92            log_info(f"Driver iniciado para o número {self.phone_number}")
 93        except Exception as e:
 94            log_error(f"Erro ao iniciar o driver para {self.phone_number}: {e}")
 95            raise

 97    def _load_cookies(self) -> None:
 98        """
 99        Carrega os cookies salvos de uma sessão anterior, se disponíveis.
100        """
101        cookies_file = self.profile_path / (self.phone_number + settings.COOKIES_FILENAME)
102        if cookies_file.exists() and self.driver:
103            try:
104                self.driver.delete_all_cookies()
105                with open(cookies_file, "rb") as file:
106                    cookies = pickle.load(file)
107                for cookie in cookies:
108                    try:
109                        self.driver.add_cookie(cookie)
110                    except Exception as e:
111                        log_error(f"Erro ao adicionar cookie: {e}")
112                log_info("Cookies carregados com sucesso.")
113            except Exception as e:
114                log_error(f"Erro ao carregar cookies: {e}")

116    def _save_cookies(self) -> None:
117        """
118        Salva os cookies atuais da sessão em um arquivo para uso futuro.
119        """
120        if not self.driver:
121            log_error("Driver não inicializado. Não é possível salvar cookies.")
122            return
123        try:
124            cookies = self.driver.get_cookies()
125            self.profile_path.mkdir(parents=True, exist_ok=True)
126            cookies_file = self.profile_path / (self.phone_number + settings.COOKIES_FILENAME)
127            with open(cookies_file, "wb") as file:
128                pickle.dump(cookies, file)
129            log_info("Cookies salvos para uso futuro.")
130        except Exception as e:
131            log_error(f"Erro ao salvar cookies: {e}")
132            raise

168    def ensure_logged_in(self) -> dict:
169        """
170        Garante que a sessão do WhatsApp esteja autenticada.
171
172        Caso existam cookies/metadados salvos e a sessão esteja ativa, utiliza-os para evitar um novo login.
173        Se a página apresentar o QR Code para login manual, captura o QR Code e retorna imediatamente
174        para o usuário, mantendo a sessão aberta.
175
176        :return: Dicionário com o status da sessão.
177                Exemplo: {"status": "logged_in"} ou {"status": "qr_required", "qr_code": "<base64>"}
178        :raises Exception: Se ocorrer um erro crítico na autenticação.
179        """
180        self._setup_driver()
181        self.driver.get("https://web.whatsapp.com/")
182        
183        # Se existirem cookies salvos, carrega-os e atualiza a página.
184        cookies_file = self.profile_path / (self.phone_number + settings.COOKIES_FILENAME)
185        if cookies_file.exists():
186            self._load_cookies()
187            self.driver.refresh()
188        
189        try:
190            # Aguarda até 10 segundos para que apareça OU o painel autenticado OU o QR Code.
191            WebDriverWait(self.driver, self.wait_time).until(
192                lambda d: d.find_elements(By.XPATH, settings.LOGGED_IN) or d.find_elements(By.XPATH, settings.QR_CODE)
193            )
194            # Agora, verifica qual elemento foi encontrado.
195            if self.driver.find_elements(By.XPATH, settings.LOGGED_IN):
196                log_info("Sessão já autenticada. Utilizando cookies/metadados salvos.", name="WhatsAppSession")
197                self._save_cookies()
198                self._save_metadata()
199                return {"status": "logged_in"}
200            elif self.driver.find_elements(By.XPATH, settings.QR_CODE):
201                log_info("Login page detectada (QR Code exibido).", name="WhatsAppSession")
202                # Captura o QR Code imediatamente e retorna o caminho do arquivo.
203                qr_file = self.capture_qr_code_to_file(self.phone_number + "_qr_code.png")
204                return {"status": "qr_required", "qr_code": qr_file}
205            else:
206                raise Exception("Nenhum elemento esperado foi encontrado.")
207        except Exception as e:
208            log_error(f"Erro ao verificar status de login: {e}", name="WhatsAppSession")
209            raise

212    def update_login_status(self) -> dict:
213        """
214        Verifica se a sessão já foi autenticada, atualizando o QR Code se necessário.
215        Se a sessão não estiver autenticada, captura o QR Code novamente.
216
217        :return: Dicionário com o status atualizado.
218                Exemplo: {"status": "logged_in"} ou {"status": "qr_required", "qr_code": "<base64>"}
219        """
220        try:
221            # Tenta aguardar por um curto intervalo (por exemplo, 5 segundos)
222            WebDriverWait(self.driver, 20).until(
223                EC.presence_of_element_located((By.XPATH, settings.LOGGED_IN))
224            )
225            log_info("Login efetuado com sucesso.", name="WhatsAppSession")
226            self._save_cookies()
227            self._save_metadata()
228            return {"status": "logged_in"}
229        except Exception:
230            try:
231                qr_code = self.capture_qr_code_to_file(self.phone_number + "_qr_code.png")  # Captura nova versão do QR Code
232            except Exception as e:
233                log_error(f"Erro ao atualizar QR Code: {e}", name="WhatsAppSession")
234                return {"status": "error", "error": str(e)}
235            return {"status": "qr_required", "qr_code": qr_code}

239        """

256        """
257        Altera o proxy utilizado na sessão.
258        
259        Encerra o driver atual, atualiza os metadados com o novo proxy, e reinicializa o driver.
260        
261        :param new_proxy: Novo proxy a ser utilizado (ex.: "http://user:pass@proxy_server:port").

263        log_info(f"Alterando proxy para {new_proxy}")
264        self.proxy = new_proxy
265        self._save_metadata()
266        self.close()
267        self._setup_driver()
268        # Após reinicialização, recarrega cookies para preservar a sessão, se possível.
269        self.driver.get("https://web.whatsapp.com/")
270        sleep(5)
271        self._load_cookies()
272        self.driver.refresh()

274    def destroy_session(self) -> None:
275        """
276        Destrói a sessão atual.
277        
278        Encerra o driver e remove todo o diretório da sessão (cookies, metadados, etc.).
279        Cuidado: essa operação é irreversível.
280        """
281        self.close()
282        try:
283            if self.profile_path.exists():
284                for item in self.profile_path.iterdir():
285                    item.unlink()
286                self.profile_path.rmdir()
287                log_info(f"Dados da sessão '{self.phone_number}' removidos com sucesso.")
288        except Exception as e:
289            log_error(f"Erro ao destruir a sessão '{self.phone_number}': {e}")
290
291    def close(self) -> None:
292        """
293        Encerra a sessão do driver.
294        """
295        if self.driver:
296            try:
297                self.driver.quit()
298                log_info(f"Driver encerrado para o número {self.phone_number}")
299            except Exception as e:
300                log_error(f"Erro ao encerrar o driver para {self.phone_number}: {e}")
301            finally:
302                self.driver = None

305    def capture_qr_code_to_file(self, filename: str = "qr_code.png") -> str:
306        """
307        Captura o QR Code exibido na página de login do WhatsApp (elemento canvas)
308        e salva a imagem em um arquivo PNG na pasta definida.
309        
310        :param filename: Nome do arquivo onde a imagem será salva.
311        :return: Caminho absoluto do arquivo salvo.
312        :raises Exception: Se não for possível localizar ou capturar o QR Code.
313        """
314        try:
315            # Define a pasta onde os QR Codes serão salvos.
316            qr_folder = Path(settings.QR_CODE_DIR) if hasattr(settings, "QR_CODE_DIR") else Path("qrcodes")
317            qr_folder.mkdir(parents=True, exist_ok=True)
318            
319            # Gera o caminho completo para o arquivo
320            file_path = qr_folder / filename
321
322            # Usa a condição de visibilidade para garantir que o canvas esteja renderizado.
323            canvas = self.driver.find_element(By.XPATH, settings.QR_CODE)
324            png_data = canvas.screenshot_as_png
325            with open(file_path, "wb") as f:
326                f.write(png_data)
327            log_info(f"QR Code salvo em '{file_path}'.", name="WhatsAppSession")
328            return str(file_path)
329        except Exception as e:
330            log_error(f"Erro ao capturar QR Code: {e}", name="WhatsAppSession")
331            raise

169    def ensure_logged_in(self) -> dict:
170        """
171        Garante que a sessão do WhatsApp esteja autenticada.
172
173        Caso existam cookies/metadados salvos e a sessão esteja ativa, utiliza-os para evitar um novo login.
174        Se a página apresentar o QR Code para login manual, captura o QR Code e retorna imediatamente
175        para o usuário, mantendo a sessão aberta.
176
177        :return: Dicionário com o status da sessão.
178                Exemplo: {"status": "logged_in"} ou {"status": "qr_required", "qr_code": "<base64>"}
179        :raises Exception: Se ocorrer um erro crítico na autenticação.
180        """
181        self._setup_driver()
182        self.driver.get("https://web.whatsapp.com/")
183        
184        # Se existirem cookies salvos, carrega-os e atualiza a página.
185        cookies_file = self.profile_path / (self.phone_number + settings.COOKIES_FILENAME)
186        if cookies_file.exists():
187            self._load_cookies()
188            self.driver.refresh()
189        
190        try:
191            # Aguarda até 10 segundos para que apareça OU o painel autenticado OU o QR Code.
192            WebDriverWait(self.driver, self.wait_time).until(
193                lambda d: d.find_elements(By.XPATH, settings.LOGGED_IN) or d.find_elements(By.XPATH, settings.QR_CODE)
194            )
195            # Agora, verifica qual elemento foi encontrado.
196            if self.driver.find_elements(By.XPATH, settings.LOGGED_IN):
197                log_info("Sessão já autenticada. Utilizando cookies/metadados salvos.", name="WhatsAppSession")
198                self._save_cookies()
199                self._save_metadata()
200                return {"status": "logged_in"}
201            elif self.driver.find_elements(By.XPATH, settings.QR_CODE):
202                log_info("Login page detectada (QR Code exibido).", name="WhatsAppSession")
203                # Captura o QR Code imediatamente e retorna o caminho do arquivo.
204                qr_file = self.capture_qr_code_to_file(self.phone_number + "_qr_code.png")
205                return {"status": "qr_required", "qr_code": qr_file}
206            else:
207                raise Exception("Nenhum elemento esperado foi encontrado.")
208        except Exception as e:
209            log_error(f"Erro ao verificar status de login: {e}", name="WhatsAppSession")
210            raise
